127 typedef enum
128 {
129  // MPF vc mapping BBB feature header (read)
130  CCI_MPF_VC_MAP_CSR_DFH = 0,
131  // BBB feature ID low (read)
132  CCI_MPF_VC_MAP_CSR_ID_L = 8,
133  // BBB feature ID high (read)
134  CCI_MPF_VC_MAP_CSR_ID_H = 16,
135 
136  //
137  // eVC_VA to physical channel mapping configuration.
138  //
139  // Groups are controlled individually. The bits are read in a given
140  // group only when the group enable bit is set in the high bits of
141  // the CSR.
142  //
143  // GROUP A:
144  //
145  // Bit 0:
146  // Enable mapping when 1 on c0 for reads. Default 1.
147  // Writing 0 will disable mapping and requests on eVC_VA will be
148  // forwarded to the FIU remaining on eVC_VA.
149  // Bit 1:
150  // The equivalent of bit 0, but on c1 for writes.
151  // Writing 0 will disable mapping and requests on eVC_VA will be
152  // forwarded to the FIU remaining on eVC_VA.
153  // Bit 2:
154  // Enable dynamic mapping. Default 1.
155  // Dynamic mapping monitors traffic and adjusts the ratio of
156  // mapped channels to optimize throughput. Optimal mapping
157  // varies depending on the ratio of reads to writes and the
158  // number of lines per request.
159  // Bits 6-3:
160  // Log2 of the dynamic sampling window size in cycles. The
161  // dynamic mapper will consider changing only after 16
162  // consecutive windows suggest the same ratio.
163  //
164  // GROUP B:
165  //
166  // Bit 7:
167  // When 0: Only change the mapping on incoming requests to eVC_VA.
168  // When 1: Treat all incoming requests as though they were on eVC_VA.
169  // All requests will then be mapped using whatever policy
170  // is set for eVC_VA.
171  //
172  // GROUP C:
173  //
174  // Bit 8:
175  // When 0: Set mapping ratio to default for the platform.
176  // When 1: Use the ratio specifiers in bits 14-9.
177  // Bits 15-9:
178  // When bit 8 is 1 these are used to define the fraction of requests
179  // mapped to channel VL0. The value is the number of 64ths that
180  // should be assigned VL0. The remaining channel mappings are split
181  // evenly between VH0 and VH1.
182  //
183  // GROUP D:
184  //
185  // Bit 31-16:
186  // The traffic threshold in a sampling window below which all
187  // requests should be directed to the low latency VL0 port. Low
188  // traffic does not need the bandwidth of multiple memory ports
189  // and may depend on low latency. (E.g. pointer chasing)
190  //
191  // The value here is the sum of all read and write lines requested
192  // in a sampling window.
193  //
194  //
195  // GROUP CONTROL:
196  //
197  // Bit 60: Group D enable
198  // Bit 61: Group C enable
199  // Bit 62: Group B enable
200  // Bit 63: Group A enable
201  //
202  CCI_MPF_VC_MAP_CSR_CTRL_REG = 24,
203 
204  // Statistics -- all 8 byte read-only CSRs
205  CCI_MPF_VC_MAP_CSR_STAT_NUM_MAPPING_CHANGES = 32,
206 
207  // Mapping history. A vector of 8 bit values with the most recent state
208  // in the low bits. Each 8 bit value is the ratio of VL0 requests in
209  // 64ths.
210  CCI_MPF_VC_MAP_CSR_STAT_HISTORY = 40,
211 
212  // Must be last
213  CCI_MPF_VC_MAP_CSR_SIZE = 48
214 }
