178 typedef struct
179 {
180  // Hits and misses in the TLB. The VTP pipeline has local caches
181  // within the pipeline itself that filter requests to the TLB.
182  // The counts here increment only for requests to the TLB service
183  // that are not satisfied in the VTP pipeline caches.
184  uint64_t numTLBHits4KB;
185  uint64_t numTLBMisses4KB;
186  uint64_t numTLBHits2MB;
187  uint64_t numTLBMisses2MB;
188 
189  // Number of cycles spent with the page table walker active. Since
190  // the walker manages only one request at a time the latency of the
191  // page table walker can be computed as:
192  // numPTWalkBusyCycles / (numTLBMisses4KB + numTLBMisses2MB)
193  uint64_t numPTWalkBusyCycles;
194 
195  // Number of failed virtual to physical translations. The VTP page
196  // table walker currently goes into a terminal error state when a
197  // translation failure is seen and blocks all future requests.
198  // If this number is non-zero the program has passed an illegal
199  // address and should be fixed.
200  uint64_t numFailedTranslations;
201  // Last virtual address translated. If numFailedTranslations is non-zero
202  // this is the failing virtual address.
203  void* ptWalkLastVAddr;
204 }
